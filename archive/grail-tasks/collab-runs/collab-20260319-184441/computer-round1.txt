"""CW congruence analysis and Lucas theorem check."""

def comb(n, k):
    if k < 0 or k > n: return 0
    if k == 0 or k == n: return 1
    k = min(k, n - k)
    r = 1
    for i in range(k): r = r * (n - i) // (i + 1)
    return r
